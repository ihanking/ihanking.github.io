#通过sklearn的数据预处理方法对缺失值进行处理
nan_model=Imputer(missing_values='NaN',strategy='mean',axis=0) #建立替换规则：将值为NaN的缺失值以均值做替换
nan_result=nan_model.fit_transform(df) #应用模型规则

#打印输出
print(nan_result)
