profit=int(input('Show me the money: '))

rates=[0.1,0.075,0.05,0.03,0.015,0.01]

    if profit<=thresholds[i]:
        bonus+=profit*rates[i]
        profit=0
        break
    else:
        bonus+=thresholds[i]*rates[i]
        profit-=thresholds[i]
bonus+=profit*rates[-1]
