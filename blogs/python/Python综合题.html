5.用函数实现如下要求：

（2）判断这20个学生成绩的等级（A:大于90小于等于100，B：大于80小于等于90，C：80及以下）

答案:

import random

def get_level(score):
  if 90 < score <= 100:
    return 'A'
  elif 80 < score <= 90:
    return 'B'
  else:
    return 'C'

def main():
  for i in range(20):
    score = random.randint(1,100)
    print("成绩为%s,等级为%s" %(score,get_level(score)))

main()
